def tree_to_s(first_call = true , nodes = [])
  nodes << Node.new(self)
  get_children.each do |child|
    nodes << child.tree_to_s(false, nodes)
  end
  if first_call
    "\n" + nodes.compact.map{|node| node.to_s}.join("\n") + "\n\n"
  end
end
