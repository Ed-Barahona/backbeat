def self.schedule(options = {}, run_at = Time.now)
  job = new(options[:event].id, options[:method], options[:args], options[:max_attempts])
  Delayed::Job.enqueue(job, run_at: run_at)
end

    event.update_status!(error, :async_job_error)
  end
