def feed(subject, name = :timeout)
  dog = Watchdog.where(subject: subject, name: name).first
  if dog
    dog.feed
  else
    Watchdog.start(subject,name)
  end
end

def start(subject, name = :timeout, starves_in = 15.minutes)
  Watchdog.dismiss(subject,name)
  new_dog = create!(name: name,
                    starves_in: starves_in,
                    subject: subject)
  new_dog.timer = Delayed::Backend::Mongoid::Job.enqueue(new_dog, run_at: new_dog.starves_in.from_now)
  new_dog.save!
  new_dog
end

def feed
  if self.timer
    self.timer.update_attributes!(run_at: self.starves_in.from_now)
  else
    self.update_attributes!(timer: Delayed::Backend::Mongoid::Job.enqueue(self, run_at: self.starves_in.from_now))
  end
end
